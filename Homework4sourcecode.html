<script>
    let chart, meanChart, varianceChart;
    let simulationData = [];
    let simulationInterval;
    let simulationTime = 0;
    let timeLimit = 0;
    let overallMeans = [];
    let overallVariances = [];

    function randomColor() {
        return 'rgba(' + Math.floor(Math.random() * 255) + ',' + Math.floor(Math.random() * 255) + ',' + Math.floor(Math.random() * 255) + ',1)';
    }

    function startSimulation() {
        stopSimulation(); // Stop any previous simulation if running

        simulationData = [];
        simulationTime = 0;
        overallMeans = [];
        overallVariances = [];
        if (chart) chart.destroy();
        if (meanChart) meanChart.destroy();
        if (varianceChart) varianceChart.destroy();

        const numAttackers = parseInt(document.getElementById("attackers").value);
        const lambda = parseFloat(document.getElementById("lambda").value);
        const n = parseInt(document.getElementById("subdivision").value);
        const dt = 1 / n; // Infinitesimal time step (subdivision of time)
        timeLimit = parseInt(document.getElementById("timeLimit").value) * 1000; // Convert seconds to milliseconds

        for (let i = 0; i < numAttackers; i++) {
            simulationData.push({
                label: `Attacker ${i + 1}`,
                data: [0],
                borderColor: randomColor(),
                fill: false
            });
        }

        const labels = [0];

        const ctx1 = document.getElementById('simulationChart').getContext('2d');
        chart = new Chart(ctx1, {
            type: 'line',
            data: {
                labels: labels,
                datasets: simulationData.map(attacker => ({
                    label: attacker.label,
                    data: attacker.data,
                    borderColor: attacker.borderColor,
                    fill: false
                }))
            },
            options: {
                scales: {
                    x: {
                        title: {
                            display: true,
                            text: 'Time (ticks)',
                            color: '#39FF14'
                        },
                        ticks: {
                            color: '#39FF14'
                        }
                    },
                    y: {
                        title: {
                            display: true,
                            text: 'Successes',
                            color: '#39FF14'
                        },
                        ticks: {
                            color: '#39FF14'
                        },
                        beginAtZero: true,
                        grid: {
                            display: true,
                            color: '#444'
                        }
                    }
                },
                responsive: true
            }
        });

        let tick = 1;
        simulationInterval = setInterval(() => {
            simulationTime += (dt * 1000);

            if (simulationTime >= timeLimit) {
                stopSimulation();
                return;
            }

            labels.push(tick);

            simulationData.forEach((attacker, i) => {
                let jump = Math.random() < (lambda * dt) ? 1 : 0;
                let lastSuccess = attacker.data[attacker.data.length - 1];
                attacker.data.push(lastSuccess + jump);
            });

            chart.update();
            tick++;
        }, dt * 1000);
    }

    function stopSimulation() {
        clearInterval(simulationInterval);
    }
</script>
